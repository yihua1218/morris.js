Bar charts

// Use Morris.Bar
Morris.Bar({
  element: 'graph',
  data: [
    {x: '2011 Q1', y: 0, z: 0, a: 0},
    {x: '2011 Q2', y: 0, z: null, a: 0},
    {x: '2011 Q3', y: 0, z: 25, a: 0},
    {x: '2011 Q4', y: 0, z: 0, a: 0}
  ],
  xkey: 'x',
  ymax: 'auto 20',
  yLabelFormat: (y) => { return Math.round(y); },
  ykeys: ['y', 'z', 'a'],
  labels: ['Y', 'Z', 'A']
}).on('click', function(i, row){
  console.log(i, row);
});
